{# Crear Servicio #}

Endpoint: POST /api/containers/

Ejemplo:

curl --request POST \
  {{ request.scheme }}://{{ request.get_host }}/api/containers/ \
  --header 'Authorization: Bearer {{ token }}' \
  --header 'Content-Type: application/json' \
  --data '{
    "name": "mi-nginx",
    "image": "nginx:latest",
    "mode": "default",
    "project": 1,
    "subject": 1
  }'

Respuesta esperada:

{
  "id": 42,
  "name": "mi-nginx",
  "image": "nginx:latest",
  "status": "creating",
  "assigned_port": 45001,
  "created_at": "2025-01-14T22:00:00Z"
}

{# Ejemplo de CI/CD #}
Ejemplo de Integración con GitHub Actions
# .github/workflows/deploy.yml
name: Deploy to PaaSify

on:
  push:
    branches: [main]

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      
      - name: Deploy to PaaSify
        run: |
          curl --request POST \
            {{ request.scheme }}://{{ request.get_host }}/api/containers/ \
            --header 'Authorization: Bearer ${{ '{{' }} secrets.PAASIFY_TOKEN {{ '}}' }}' \
            --header 'Content-Type: application/json' \
            --data '{
              "name": "my-app",
              "image": "nginx:latest",
              "mode": "default",
              "project": 1,
              "subject": 1
            }'
